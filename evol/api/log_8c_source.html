49 static void
50 stdout_callback(log_Event *ev)
51 {
52  char buf[16];
53  buf[strftime(buf, sizeof(buf), "%H:%M:%S", ev->time)] = '\0';
54 #ifdef LOG_USE_COLOR
55  fprintf(ev->udata,
56  "%s %s%-5s\x1b[0m \x1b[90m%s:%d:\x1b[0m ",
57  buf,
58  level_colors[ev->level],
59  level_strings[ev->level],
60  ev->file,
61  ev->line);
62 #else
63  fprintf(
64  ev->udata, "%s %-5s %s:%d: ", buf, level_strings[ev->level], ev->file, ev->line);
65 #endif
66  vfprintf(ev->udata, ev->fmt, ev->ap);
67  fprintf(ev->udata, "\n");
68  fflush(ev->udata);
69 }

152 void
153 log_log(int level, const char *file, int line, const char *fmt, ...)
154 {
155  log_Event ev = {
156  .fmt = fmt,
157  .file = file,
158  .line = line,
159  .level = level,
160  };
161 
162  lock();
163 
164  if (!L.quiet && level >= L.level) {
165  init_event(&ev, stderr);
166  va_start(ev.ap, fmt);
167  stdout_callback(&ev);
168  va_end(ev.ap);
169  }
170 
171  for (int i = 0; i < MAX_CALLBACKS && L.callbacks[i].fn; i++) {
172  Callback *cb = &L.callbacks[i];
173  if (level >= cb->level) {
174  init_event(&ev, cb->udata);
175  va_start(ev.ap, fmt);
176  cb->fn(&ev);
177  va_end(ev.ap);
178  }
179  }
180 
181  unlock();
182 }
